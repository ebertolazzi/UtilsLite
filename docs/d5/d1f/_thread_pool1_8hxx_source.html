63 class Worker {
64
65 std::mutex m_task_mutex;
66 std::condition_variable m_task_cv;
67
68 bool m_active{true};
69 std::thread m_running_thread;
70 bool m_task_done{true};
71 FUN m_task;
72
74 void
75 worker_loop() {
76 std::unique_lock<std::mutex> lock(m_task_mutex);
77 while ( m_active ) {
78 while ( m_task_done ) m_task_cv.wait( lock );
79 //m_task_cv.wait( lock, [this] { return !m_task_done; } );
80 m_task();
81 m_task_done = true;
82 m_task_cv.notify_one();
83 }
84 }
85
86 public:
87
88 //disable copy

90 //Worker( Worker && ) = delete;
91 Worker& operator = ( Worker const & ) = delete;
92 Worker& operator = ( Worker && ) = delete;
93
97 Worker() : m_running_thread( std::thread( &Worker::worker_loop, this ) ) {}
98
102 ~Worker() {
103 if ( m_active ) {
104 auto dummy_task = []()->void{};
105 m_active = false; // deactivate computation
106 exec( dummy_task );
107 if ( m_running_thread.joinable() ) m_running_thread.join(); // wait thread for exiting
108 }
109 }
110
114 Worker( Worker && rhs ) noexcept
115 : m_active(rhs.m_active)
116 , m_running_thread(std::move(rhs.m_running_thread))
117 , m_task(std::move(rhs.m_task))
118 {}

123 void
124 wait() {
125 std::unique_lock<std::mutex> lock(m_task_mutex);
126 while ( !m_task_done ) m_task_cv.wait( lock );
127 //m_task_cv.wait( lock, [this] { return m_task_done; });
128 }
129
135 void
136 exec( FUN && fun ) {
137 // aspetto che pointer si liberi se occupato
138 std::unique_lock<std::mutex> lock(m_task_mutex);
139 while ( !m_task_done ) m_task_cv.wait( lock );
140 //m_task_cv.wait( lock, [this] { return m_task_done; } );
141 m_task_done = false;
142 m_task = std::move(fun); // cambia funzione da eseguire
143 m_task_cv.notify_one();
144 }
145 };
