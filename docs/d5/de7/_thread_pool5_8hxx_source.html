81 void
82 worker_loop() {
83 m_is_running.red(); // block computation
84 while ( m_active ) {
85 m_is_running.wait(); // wait signal to start computation
86 // ----------------------------------------

89 // ----------------------------------------

94 }

190 // Vector of workers managed by the thread pool.
191 std::vector<Worker> m_workers;
192 // Stack of available worker IDs.
193 std::vector<unsigned> m_stack;
194 std::mutex m_stack_mutex;
195 std::condition_variable m_stack_cond;
196
200 void setup() { for ( auto & w: m_workers ) w.start(); }

207 void
208 resize_workers( unsigned numThreads ) {
209 m_stack.clear(); // empty stack
210 m_stack.reserve( size_t(numThreads) );
211 m_workers.resize( size_t(numThreads) );
212 unsigned id = 0;
213 for ( Worker & w : m_workers ) { w.setup( this, id ); ++id; }
214 while ( id-- > 0 ) push_worker( id );
215 setup();
216 }
217
223 void
224 push_worker( unsigned id ) {
225 std::unique_lock<std::mutex> lock(m_stack_mutex);
226 m_stack.emplace_back(id);
227 m_stack_cond.notify_one();
228 }
229
235 unsigned
236 pop_worker() {
237 std::unique_lock<std::mutex> lock(m_stack_mutex);
238 m_stack_cond.wait( lock, [&]()->bool { return !m_stack.empty(); } );
239 unsigned id = m_stack.back(); m_stack.pop_back();
240 return id;
241 }
