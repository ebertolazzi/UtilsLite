109 Queue( unsigned capacity )
110 : m_queue_data( std::size_t( capacity+1 ) )
111 , m_size( capacity+1 )
112 , m_capacity( capacity )
113 { }
114
115 void
116 push( TaskData * task ) {
117 std::lock_guard<std::mutex> lock(m_mutex);
118 m_queue_data[m_push_ptr] = task;
119 if ( ++m_push_ptr == m_size ) m_push_ptr = 0;
120 }
121
122 TaskData *
123 pop() {
124 std::lock_guard<std::mutex> lock(m_mutex);
125 unsigned ipos{m_pop_ptr};
126 if ( ++m_pop_ptr == m_size ) m_pop_ptr = 0;
127 return m_queue_data[ipos];
128 }
129
130 unsigned
131 size() const {
132 std::lock_guard<std::mutex> lock(m_mutex);
133 return ((m_push_ptr + m_size) - m_pop_ptr) % m_size;
134 }
135
136 bool
137 empty() const {
138 std::lock_guard<std::mutex> lock(m_mutex);
139 return m_push_ptr == m_pop_ptr;
140 }
141
142 bool is_full() const {
143 std::lock_guard<std::mutex> lock(m_mutex);
144 unsigned sz = ((m_push_ptr + m_size) - m_pop_ptr) % m_size;
145 return sz >= m_capacity;
146 }
147
148 unsigned capacity() const { return m_capacity; }
149
151 void
152 clear() {
153 std::lock_guard<std::mutex> lock(m_mutex);
154 while( m_push_ptr != m_pop_ptr ) delete pop();
155 }
156
157 void
158 resize( unsigned capacity ) {
159 std::lock_guard<std::mutex> lock(m_mutex);
160 while( m_push_ptr != m_pop_ptr ) delete pop();
161 m_size = capacity+1;
162 m_capacity = capacity;
163 m_queue_data.resize( m_size );
164 }

166 ~Queue() = default;
