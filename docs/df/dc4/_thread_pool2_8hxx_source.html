48 using TYPE = std::function<void(void)>;
49
50 std::atomic_int m_jobs_left{0};
51 std::atomic_bool m_bailout{false};
52 std::atomic_bool m_finished{false};
53
54 std::condition_variable m_job_available_var;
55 std::condition_variable m_wait_var;
56 std::mutex m_wait_mutex;
57 std::mutex m_queue_mutex;
58
59 std::list<TYPE> m_queue;
60 std::vector<std::thread> m_threads;

66 void
67 Task() {
68 TYPE job;
69 while( !m_bailout ) {
70 {
71 std::unique_lock<std::mutex> job_lock( m_queue_mutex );
72 while ( m_queue.empty() && !m_bailout ) m_job_available_var.wait( job_lock );
73 if( !m_bailout ) { job = m_queue.front(); m_queue.pop_front(); }
74 }
75 if ( !m_bailout ) { job(); --m_jobs_left; }
76 {
77 std::lock_guard<std::mutex> wait_lock( m_wait_mutex );
78 m_wait_var.notify_one();
79 }
80 }
81 }

123 void
124 join() override {
125 if( m_finished ) return;
126 wait();
127 // note that we're done, and wake up any thread that's
128 // waiting for a new job
129 m_bailout = true;
130 m_job_available_var.notify_all();
131 for( auto & t : m_threads )
132 if( t.joinable() )
133 t.join();
134 m_finished = true;
135 }
