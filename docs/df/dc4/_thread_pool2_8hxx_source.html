207 /*\
208 This queue requires attention for protection against
209 concurrent access. Protect against:
210
211 - Concurrent access by two worker threads both wanting to get()
212 a task from the queue at the same time.
213
214 - Concurrent access by two threads both wanting to
215 put() a task into the queue at the same time.
216
217 - A worker thread having determined that the queue is empty,

273 template<class C>
274 void
275 put( C&& c ) {
276 std::unique_lock<std::mutex> lock(m_push_mutex);
277 while ( m_queue.is_full() ) {
278 // No space in the queue. Must wait for workers to advance.
279 lock.unlock();
280 try_help( m_queue.capacity() / 2 );
281 lock.lock();
282 }
283 // Now there is space in the queue and we have locked the back.
284
285 // Enqueue function.
286 if ( m_shutting_down ) {
287 QueueElement fun(std::forward<C>(c)); // Run Function destructor
288 } else {
289 // Here we have exclusive access to the head of the queue.
290 m_queue.push(std::forward<C>(c));
291
292 if ( m_idle_workers && !m_wakeup_is_pending ) {
293 m_wakeup_is_pending = true;
294 m_waiting_workers_cond.notify_one(); // wake up a worker
295 }
296 }
