196 inline
197 int64_t
198 mex_get_int64( mxArray const * arg, char const msg[] ) {
199 mwSize number_of_dimensions = mxGetNumberOfDimensions(arg);
200 UTILS_MEX_ASSERT0( number_of_dimensions == 2, msg );
201 mwSize const * dims = mxGetDimensions(arg);
202 UTILS_MEX_ASSERT(
203 dims[0] == 1 && dims[1] == 1,
204 "{}, found {} x {} matrix\n",
205 msg, dims[0], dims[1]
206 );
207 mxClassID category = mxGetClassID(arg);
208 int64_t res = 0;
209 void *ptr = mxGetData(arg);
210 switch (category) {
211 case mxINT8_CLASS: res = *static_cast<uint8_t*>(ptr); break;
212 case mxUINT8_CLASS: res = *static_cast<uint8_t*>(ptr); break;
213 case mxINT16_CLASS: res = *static_cast<int16_t*>(ptr); break;
214 case mxUINT16_CLASS: res = *static_cast<uint16_t*>(ptr); break;
215 case mxINT32_CLASS: res = *static_cast<int32_t*>(ptr); break;
216 case mxUINT32_CLASS: res = *static_cast<uint32_t*>(ptr); break;
217 case mxINT64_CLASS: res = *static_cast<int64_t*>(ptr); break;
218 case mxUINT64_CLASS: res = *static_cast<uint64_t*>(ptr); break;
219 case mxDOUBLE_CLASS:
220 { double tmp = *static_cast<double*>(ptr);
221 UTILS_MEX_ASSERT( tmp == std::floor(tmp), "{} expected int, found {}\n", msg, tmp );
222 res = static_cast<int64_t>(tmp);
223 }
224 break;
225 case mxSINGLE_CLASS:
226 { float tmp = *static_cast<float*>(ptr);
227 UTILS_MEX_ASSERT( tmp == std::floor(tmp), "{} expected int, found {}\n", msg, tmp );
228 res = static_cast<int64_t>(tmp);
229 }
230 break;
231 default:
232 UTILS_MEX_ASSERT( false, "{} bad type scalar", msg );
233 break;
234 }
235 return res;
236 }

471 template <typename R, typename I>
472 inline
473 int
474 mex_create_sparse_matrix(
475 size_t nnz,
476 size_t nrows,
477 size_t ncols,
478 I i_rows[],
479 I j_cols[],
480 R vals[],
481 mxArray * arg_out[]
482 ) {
483 mxArray *args[5]; // Array of arguments to be passed to MATLAB's sparse function.
484
485 // Create MATLAB matrices to store row indices, column indices, and values.
486 double *Irow = mex_create_matrix_value(args[0], 1, nnz);
487 double *Jcol = mex_create_matrix_value(args[1], 1, nnz);
488 double *VALS = mex_create_matrix_value(args[2], 1, nnz);
489
490 // Set the number of rows and columns in the sparse matrix.
491 mex_set_scalar_value(args[3], nrows);
492 mex_set_scalar_value(args[4], ncols);
493
494 // Convert the row and column indices to 1-based indexing for MATLAB.
495 std::transform(i_rows, i_rows + nnz, Irow, [](I val) -> double { return double(val + 1); });
496 std::transform(j_cols, j_cols + nnz, Jcol, [](I val) -> double { return double(val + 1); });
497
498 // Copy the values of the non-zero elements into the MATLAB array.
499 std::copy_n(vals, nnz, VALS);
500
501 // Call the MATLAB function 'sparse' to create the sparse matrix.
502 return mexCallMATLAB(1, arg_out, 5, args, "sparse");
503 }

541 template <typename base>
542 class mex_class_handle {
543 uint32_t m_signature{CLASS_HANDLE_SIGNATURE};
544 base * m_ptr{nullptr};
545 std::string m_name;
546
547 public:
548
554 mex_class_handle<base> const & operator = ( mex_class_handle<base> const & ) = delete;
555
561 mex_class_handle() = delete;
562
568 explicit
569 mex_class_handle( base * ptr )
570 : m_ptr(ptr)
571 , m_name(typeid(base).name())
572 {}
573
577 ~mex_class_handle()
578 { m_signature = 0; delete m_ptr; m_ptr = nullptr; }
579
587 bool is_valid()
588 { return ((m_signature == CLASS_HANDLE_SIGNATURE) &&
589 !strcmp(m_name.c_str(), typeid(base).name())); }
590
596 base * ptr() { return m_ptr; }
597 };

612 mex_convert_ptr_to_mx( base * ptr ) {
613 mexLock();
614 mxArray *out = mxCreateNumericMatrix(1, 1, mxUINT64_CLASS, mxREAL);
615 *((uint64_t *)mxGetData(out)) = reinterpret_cast<uint64_t>(
616 new mex_class_handle<base>(ptr)
617 );
618 return out;
619 }

632 template <typename base>
633 inline
634 mex_class_handle<base> *
635 mex_convert_mx_to_handle_ptr( mxArray const * in ) {
636 if ( mxGetNumberOfElements(in) != 1 ||
637 mxGetClassID(in) != mxUINT64_CLASS ||
638 mxIsComplex(in) ) {
639 mexErrMsgTxt( "Input must be an uint64 scalar." );
640 }
641 mex_class_handle<base> * ptr = reinterpret_cast<mex_class_handle<base> *>(
642 *((uint64_t *)mxGetData(in))
643 );
644 if ( !ptr->is_valid() ) mexErrMsgTxt( "Handle not valid." );
645 return ptr;
646 }
