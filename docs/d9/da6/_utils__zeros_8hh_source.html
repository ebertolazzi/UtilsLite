248 Real t1 = s*s;
249 Real t2 = t1*t1;
250 Real t17 = t*t;
251 Real t22 = u*u;
252 Real t32 = t17*t17;
253 Real t34 = t*t32;
254 Real t37 = t*t17;
255 Real t46 = 1+v;

258 return (2*t-1)*(2+5*t)*u*t*t2 +
259 (4*t+1)*u*s*t2+
